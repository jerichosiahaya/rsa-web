create table pelanggan (
    id int PRIMARY KEY,
    nama varchar(250),
    alamat varchar(250));

create table mobil (
    noRangka varchar(250) PRIMARY KEY,
    noPolisi varchar(250),
    tglBeli date,
    id int,
    FOREIGN KEY (id) REFERENCES pelanggan(id))

create table detail_servis (
    idServis int PRIMARY KEY,
    tglServisTerakhir date,
    tglServisSelanjutnya date,
    noRangka varchar(250),
    FOREIGN KEY (noRangka) REFERENCES mobil(noRangka))
